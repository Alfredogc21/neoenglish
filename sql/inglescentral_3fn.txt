============================================================
3) TABLAS CATALOGO
============================================================

CREATE TABLE niveles (
    nivel_id INT AUTO_INCREMENT PRIMARY KEY,
    codigo_nivel VARCHAR(20) NOT NULL UNIQUE,
    nombre_nivel VARCHAR(120) NOT NULL,
    banda_cefr VARCHAR(10) NOT NULL,
    descripcion_nivel VARCHAR(300) NOT NULL,
    meta_xp INT NOT NULL CHECK (meta_xp >= 0)
);

CREATE TABLE habilidades (
    habilidad_id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_habilidad VARCHAR(90) NOT NULL UNIQUE
);

CREATE TABLE tipos_pregunta (
    tipo_pregunta_id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_tipo_pregunta VARCHAR(90) NOT NULL UNIQUE
);

============================================================
4) TABLAS ACADEMICAS
============================================================

CREATE TABLE niveles_habilidades (
    nivel_id INT NOT NULL,
    habilidad_id INT NOT NULL,
    PRIMARY KEY (nivel_id, habilidad_id),
    CONSTRAINT fk_niveles_habilidades_nivel
        FOREIGN KEY (nivel_id) REFERENCES niveles(nivel_id)
        ON DELETE CASCADE,
    CONSTRAINT fk_niveles_habilidades_habilidad
        FOREIGN KEY (habilidad_id) REFERENCES habilidades(habilidad_id)
        ON DELETE RESTRICT
);

CREATE TABLE lecciones (
    leccion_id INT AUTO_INCREMENT PRIMARY KEY,
    nivel_id INT NOT NULL,
    titulo_leccion VARCHAR(160) NOT NULL,
    objetivo_leccion VARCHAR(300) NOT NULL,
    orden_secuencia INT NOT NULL,
    CONSTRAINT uq_lecciones_nivel_orden UNIQUE (nivel_id, orden_secuencia),
    CONSTRAINT fk_lecciones_nivel
        FOREIGN KEY (nivel_id) REFERENCES niveles(nivel_id)
        ON DELETE CASCADE
);

CREATE TABLE preguntas (
    pregunta_id INT AUTO_INCREMENT PRIMARY KEY,
    leccion_id INT NOT NULL,
    tipo_pregunta_id INT NOT NULL,
    enunciado_pregunta VARCHAR(500) NOT NULL,
    explicacion VARCHAR(320) NULL,
    peso_dificultad DECIMAL(4,2) NOT NULL DEFAULT 1.00,
    orden_secuencia INT NOT NULL,
    CONSTRAINT uq_preguntas_leccion_orden UNIQUE (leccion_id, orden_secuencia),
    CONSTRAINT fk_preguntas_leccion
        FOREIGN KEY (leccion_id) REFERENCES lecciones(leccion_id)
        ON DELETE CASCADE,
    CONSTRAINT fk_preguntas_tipo
        FOREIGN KEY (tipo_pregunta_id) REFERENCES tipos_pregunta(tipo_pregunta_id)
        ON DELETE RESTRICT
);

CREATE TABLE opciones_pregunta (
    opcion_id INT AUTO_INCREMENT PRIMARY KEY,
    pregunta_id INT NOT NULL,
    texto_opcion VARCHAR(250) NOT NULL,
    es_correcta BOOLEAN NOT NULL DEFAULT FALSE,
    CONSTRAINT fk_opciones_pregunta
        FOREIGN KEY (pregunta_id) REFERENCES preguntas(pregunta_id)
        ON DELETE CASCADE
);

============================================================
5) TABLAS DE AUTENTICACION Y PROGRESO
============================================================

CREATE TABLE estudiantes (
    estudiante_id INT AUTO_INCREMENT PRIMARY KEY,
    nombres VARCHAR(80) NOT NULL,
    apellidos VARCHAR(80) NOT NULL,
    grupo_grado VARCHAR(20) NOT NULL,
    correo VARCHAR(150) NOT NULL UNIQUE,
    hash_contrasena VARCHAR(255) NOT NULL,
    creado_en TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE docentes (
    docente_id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_completo VARCHAR(120) NOT NULL,
    correo VARCHAR(150) NOT NULL UNIQUE,
    hash_contrasena VARCHAR(255) NOT NULL,
    creado_en TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE progreso_nivel_estudiante (
    estudiante_id INT NOT NULL,
    nivel_id INT NOT NULL,
    desbloqueado_en TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    completado_en TIMESTAMP NULL,
    mejor_porcentaje DECIMAL(5,2) NOT NULL DEFAULT 0.00,
    xp_ganada INT NOT NULL DEFAULT 0,
    PRIMARY KEY (estudiante_id, nivel_id),
    CONSTRAINT fk_progreso_estudiante
        FOREIGN KEY (estudiante_id) REFERENCES estudiantes(estudiante_id)
        ON DELETE CASCADE,
    CONSTRAINT fk_progreso_nivel
        FOREIGN KEY (nivel_id) REFERENCES niveles(nivel_id)
        ON DELETE CASCADE
);

CREATE TABLE intentos_estudiante (
    intento_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    estudiante_id INT NOT NULL,
    leccion_id INT NOT NULL,
    iniciado_en TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    finalizado_en TIMESTAMP NULL,
    porcentaje_puntaje DECIMAL(5,2) NOT NULL DEFAULT 0.00,
    xp_ganada INT NOT NULL DEFAULT 0,
    CONSTRAINT fk_intentos_estudiante
        FOREIGN KEY (estudiante_id) REFERENCES estudiantes(estudiante_id)
        ON DELETE CASCADE,
    CONSTRAINT fk_intentos_leccion
        FOREIGN KEY (leccion_id) REFERENCES lecciones(leccion_id)
        ON DELETE CASCADE
);

CREATE TABLE respuestas_intento_estudiante (
    respuesta_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    intento_id BIGINT NOT NULL,
    pregunta_id INT NOT NULL,
    opcion_seleccionada_id INT NULL,
    es_correcta BOOLEAN NOT NULL,
    CONSTRAINT fk_respuestas_intento
        FOREIGN KEY (intento_id) REFERENCES intentos_estudiante(intento_id)
        ON DELETE CASCADE,
    CONSTRAINT fk_respuestas_pregunta
        FOREIGN KEY (pregunta_id) REFERENCES preguntas(pregunta_id)
        ON DELETE CASCADE,
    CONSTRAINT fk_respuestas_opcion
        FOREIGN KEY (opcion_seleccionada_id) REFERENCES opciones_pregunta(opcion_id)
        ON DELETE SET NULL
);

11) JUSTIFICACION 3FN
============================================================

1FN:
- Todas las columnas son atomicas.
- No hay listas separadas por comas dentro de una misma columna.

2FN:
- En tablas con clave compuesta (niveles_habilidades y progreso_nivel_estudiante),
  los atributos dependen de toda la clave, no de una parte.

3FN:
- No existen dependencias transitivas entre atributos no clave.
- Ejemplos:
  * nombre_habilidad esta en habilidades, no repetida en niveles.
  * nombre_tipo_pregunta esta en tipos_pregunta, no repetida en preguntas.
  * Datos de intentos y respuestas estan separados para no duplicar informacion.

Resultado:
- Menor redundancia.
- Mejor integridad de datos.
- Consultas mas claras para analitica docente.
